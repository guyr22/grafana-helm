# templates/grafana-service.yaml
# This template creates the Service to expose the Grafana Deployment
# within the Kubernetes cluster.

apiVersion: v1
kind: Service
metadata:
  name: {{ include "grafana-postgres.fullname" . }}
  labels:
    {{- include "grafana-postgres.labels" . | nindent 4 }}
spec:
  type: {{ .Values.grafana.service.type }}
  ports:
    - port: {{ .Values.grafana.service.port }}
      targetPort: http
      protocol: TCP
      name: http
  selector:
    {{- include "grafana-postgres.selectorLabels" . | nindent 4 }}
