# templates/postgresql-service.yaml
# This template creates the main Service for PostgreSQL.
# This is the service that Grafana will use to connect to the database.

apiVersion: v1
kind: Service
metadata:
  name: {{ .Release.Name }}-postgresql
  labels:
    {{- include "grafana-postgres.labels" . | nindent 4 }}
    app.kubernetes.io/component: postgresql
spec:
  type: ClusterIP
  ports:
    - name: tcp-postgresql
      port: 5432
      targetPort: tcp-postgresql
  selector:
    {{- include "grafana-postgres.selectorLabels" . | nindent 4 }}
    app.kubernetes.io/component: postgresql
