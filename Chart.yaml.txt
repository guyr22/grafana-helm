# Chart.yaml
# This file defines the chart's metadata and its dependencies.
# We are defining PostgreSQL as a dependency, so Helm will automatically
# fetch and install it when you install this chart.

apiVersion: v2
name: grafana-postgres
description: A Helm chart for deploying Grafana and a dedicated PostgreSQL database for it.
type: application

# This is the chart version.
version: 0.1.0

# This is the version of the application being deployed.
appVersion: "10.4.2"

# Dependencies section to include the PostgreSQL chart from the Bitnami repository.
dependencies:
  - name: postgresql
    # Using a specific version range for predictability.
    version: "14.x.x"
    # The repository where the PostgreSQL chart is located.
    repository: https://charts.bitnami.com/bitnami
